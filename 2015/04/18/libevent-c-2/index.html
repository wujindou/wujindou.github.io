Protobuf已经下载好了，现在就开始封装一个简单的例子吧,参考官方的协议处理和封装以及编译成所需要的语言的代码
定义一个简单的EchoMessage.proto
内容如下： 

message EchoRequest
{
    required string message = 1; //请求字符串
};
message EchoResponse
{
    required string response = 1; //回应字符串
};

测试：
test.cpp 

#include<iostream>
#include<string>
#include "echo.pb.h"
using namespace std;
int main()
{
    string s="hello request";
    EchoRequest request ;
    request.set_message(s);
    cout<<request.message()<<endl;
    return 1;
}

client.cpp 

    #include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<unistd.h>
#include<iostream>
#include<sys/types.h>
#include<sys/socket.h>
#include<netinet/in.h>
#include <arpa/inet.h>
#include "echo.pb.h"

#define oops(msg) {\
    perror(msg); \
    exit(1); \
}  
#define SIZE 256  

#define LISTEN_PORT 9999  
using namespace std;

int main(int argc, char** argv)
{


    int socket_fd = socket(PF_INET, SOCK_STREAM, 0);
    if(socket_fd == -1)
        oops("socket initiating error...");


    struct sockaddr_in address;
    bzero(&address, sizeof(address));
    address.sin_family = AF_INET;
    address.sin_addr.s_addr = inet_addr("127.0.0.1");
    address.sin_port =  htons(LISTEN_PORT);

    int connect_result = connect(socket_fd, (struct sockaddr*)&address, sizeof(address));
    if(connect_result == -1)
        oops("connect error...");


         string data;
         EchoRequest req;
         req.set_message("wwwww");
         req.SerializeToString(&data);
        char bts[data.length()];
        strcpy(bts, data.c_str());
        ssize_t write_length = write(socket_fd, bts, sizeof(bts));
        if(write_length <= 0 )
            oops("error: write ...");


        char buf[512];
        bzero(buf, 0);
        ssize_t read_length = read(socket_fd, buf, sizeof(buf));
        if(read_length == -1)
            oops("error: read data from socket...");

        printf("the ret msg is: %s\n", buf);
    close(socket_fd);
    return EXIT_SUCCESS;
}
