Jquery简单封装插件

By Jackie wu

发表于 2015-04-17

由于开发过程是需要使用fancybox，但是已有的结构需要每次都把html代码如下copy到需要添加文档，然后需要每次复制函数。重复早轮子，
于是考虑通过jquery进行简单的封装以避免重复的造轮子工作。
新建fancybox-widget.js 

$(function(){
        var tbox =$(
        '<div id="show-message" style = "display:none"><div class = "cb50" ></div><p class = "t_center ml_25 mr_25 f14 f_red"></p><div class = "cb20" ></div><a href = "javascript:$.fancybox.close();" class = "f14 f_white fc t_center" style = "width:100px; height:30px; line-height:30px; display:block; background:#eb2f7b; border-radius:4px;" > 确认 </a> </div ></div>');
    $('body').prepend(tbox);

    });
    jQuery.showbox = function(data,that ){

    if (data && data.code == 1) {
    $("#show-message p").removeClass('f_red');
    $("#show-message p").addClass('f_green');
    } else {
    $("#show-message p").removeClass('f_green');
    $("#show-message p").addClass('f_red');
    }
    $("#show-message p").html(data.message);
    if (data && data.code == 1) {
    $.fancybox.open({
    'href': "#show-message",
    'scrolling': 'no',
    'modal': true,
    'closeBtn': true,
    'afterClose': function() {
    window.location.reload();
    return;
    },
    'autoSize': false,
    'fitToView': false,
    'width': 300,
    'height': 200,
    'padding': 0
    });

    } else {
    $.fancybox.open({
    'href': "#show-message",
    'scrolling': 'no',
    'modal': true,
    'closeBtn': true,
    'afterClose': function() {
    if (that) {
    $(that).attr("data-tag", '0');
    }
    return;
    },
    'autoSize': false,
    'fitToView': false,
    'width': 300,
    'height': 200,
    'padding': 0
    });
    }
    }
    在需要使用该插件的页面通过引入该js文件，然后调用也非常的简单： 
    $('#show').click(function() {
    var data = { code: 1, message: 'hello' };
    $.showbox(data);

    });
